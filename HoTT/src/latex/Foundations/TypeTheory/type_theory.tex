\subsection{Contexts}

Here are the basic principles for contexts in type theory.

\begin{definition}{0.1.1}
We will note \notation{$.$} \probe for the \actualDef{empty context} \probe
\end{definition}

\begin{axiom}{0.1.1}
\axiomName{empty derivation} \probe :

\actualAxiom{$rule\_empty\_derivation : \epsilon\ \text{——}\ empty\_context\ ctx$} \probe
\end{axiom}

\begin{axiom}{0.1.2}
\axiomName{context extension} \probe : 

\actualAxiom{$rule\_context\_extension : \ctx \vdash \var{A}:\U{i},\quad \var{x}\ not\ in\ dom(\ctx)\ \text{——}\ \ctx,\ \var{x}:\var{A}\ ctx$} \probe
\end{axiom}

\begin{axiom}{0.1.3}

\axiomName{variable rule} \probe : 
\actualAxiom{$rule\_variable : \ctx\ ctx,\quad \var{x}:\var{A}\ in\ \ctx\ \text{——}\ \ctx \vdash \var{x}:\var{A}$} \probe
\end{axiom}



% judgmental equalities
\subsection{Judgemental equalities}

We suppose that judgmental equality is an equivalence relation. Namely:

\begin{axiom}{0.2.1}
\axiomName{Judgmental reflexivity} \probe : 

\actualAxiom{$rule\_judg\_refl : \ctx \vdash \var{a}:\var{A}\ \text{——}\ \ctx \vdash \var{a} \equiv \var{a}:\var{A}$} \probe
\end{axiom}

\begin{axiom}{0.2.2}
\axiomName{Judgmental symmetry} \probe : 

\actualAxiom{$rule\_judg\_sym : \ctx \vdash \var{a}\equiv \var{b}:\var{A}\ \text{——}\ \ctx \vdash \var{b} \equiv \var{a}:\var{A}$} \probe
\end{axiom}

\begin{axiom}{0.2.3}
\axiomName{Judgmental transitivity} \probe : 

\actualAxiom{$rule\_judg\_trans : \ctx \vdash \var{a}\equiv \var{b}:\var{A},\quad \ctx \vdash \var{b}\equiv \var{c}:\var{A}\ \text{——}\ \ctx \vdash \var{a} \equiv \var{c}:\var{A}$} \probe
\end{axiom}

\begin{axiom}{0.2.4}
\axiomName{Judgmental typing} \probe : 

\actualAxiom{$rule\_judg\_typing : \ctx \vdash \var{a}:\var{A},\quad \ctx \vdash \var{A}\equiv \var{B}:\U{i}\ \text{——}\ \ctx \vdash \var{a}:\var{B}$} \probe
\end{axiom}

\begin{axiom}{0.2.5}
\axiomName{Judgmental typing equiv} \probe : 

\actualAxiom{$rule\_judg\_typing\_eq : \ctx \vdash \var{a}\equiv \var{b}:\var{A},\quad \ctx \vdash \var{A}\equiv \var{B}:\U{i}\ \text{——}\ \ctx \vdash \var{a}\equiv \var{b}:\var{B}$} \probe
\end{axiom}

% Function type
\subsection{Function type}

\begin{axiom}{1.1.1}
\axiomName{Function formation} \probe :

\actualAxiom{$rule\_\lambda-form : \ctx \vdash \var{A}:\U{i},\quad \ctx, \var{a}:\var{A} \vdash \var{B}:\U{i}\ \text{——}\ \ctx \vdash \var{A} \to \var{B}:\U{i}$} \probe
\end{axiom}

\begin{axiom}{1.1.2}
\axiomName{Function introduction} \probe : 

\actualAxiom{$rule\_\lambda-intro : \ctx, \var{a}:\var{A} \vdash \var{B}:\U{i}\ \text{——}\ \ctx \vdash \lambda (\var{x}:\var{A}).\var{b}:\var{A} \to \var{B}$} \probe
\end{axiom}

\begin{axiom}{1.1.3}
\axiomName{Function elimination} \probe : 

\actualAxiom{$rule\_\lambda-elim : \ctx \vdash \var{f}:\var{A}\to\var{B},\quad \ctx \vdash \var{a}:\var{A}\ \text{——}\ \ctx \vdash \var{f(a)}:\var{B}$} \probe
\end{axiom}

\begin{axiom}{1.1.4}
\axiomName{Function computation} \probe : 

\actualAxiom{$rule\_\lambda-comp : \ctx, \var{x}:\var{A} \vdash \var{b}:\var{B},\quad \ctx \vdash \var{a}:\var{A}\ \text{——}\ \ctx \vdash \text{let}\ \var{x} = \var{a}\ \text{in}\ \lambda (\var{x}:\var{A}).\var{b}:\var{B}$} \probe
\end{axiom}

\begin{axiom}{1.1.5}
\axiomName{Function uniqueness principle} \probe : 

\actualAxiom{$rule\_\lambda-uniq : \ctx \vdash \var{f}:\var{A}\to\var{B}\ \text{——}\ \ctx \vdash \var{f} \equiv \lambda (\var{x}:\var{A}).\var{f}(\var{x}):\var{A} \to \var{B}$} \probe
\end{axiom}

\end{axiom}

\subsection{Coproduct type}

\begin{axiom}{1.4.1}
\axiomName{Coproduct formation} \probe : 

\actualAxiom{$rule\_coproduct-form : \ctx \vdash \var{A}:\U{i},\quad \ctx \vdash \var{B}:\U{i}\ \text{——}\ \ctx \vdash \var{A}+\var{B}:\U{i}$} \probe
\end{axiom}

\begin{axiom}{1.4.2}
\axiomName{Coproduct introduction left} \probe : 

\actualAxiom{$rule\_coproduct-intro1 : \ctx \vdash \var{A}:\U{i},\quad \ctx \vdash \var{B}:\U{i},\quad \ctx \vdash \var{a}:\var{A}\ \text{——}\ \ctx \vdash inl(\var{a}):\var{A}+\var{B}$} \probe
\end{axiom}

\begin{axiom}{1.4.3}
\axiomName{Coproduct introduction right} \probe : 

\actualAxiom{$rule\_coproduct-intro2 : \ctx \vdash \var{A}:\U{i},\quad \ctx \vdash \var{B}:\U{i},\quad \ctx \vdash \var{b}:\var{B}\ \text{——}\ \ctx \vdash inr(\var{b}):\var{A}+\var{B}$} \probe
\end{axiom}

\begin{axiom}{1.4.4}
\axiomName{Coproduct elimination} \probe : 

\actualAxiom{$rule\_coproduct-elim : \ctx, \var{z}:\var{A}+\var{B} \vdash \var{C}:\U{i},\quad \ctx,\var{a}:\var{A} \vdash \var{c}:\var{C}(inl(\var{a})),\quad \ctx,\var{b}:\var{B} \vdash \var{d}:\var{C}(inr(\var{b})),\quad \ctx \vdash \var{e}:\var{A}+\var{B}\ \text{——}\ \ctx \vdash ind(\var{C},\var{c}(\var{a}),\var{d}(\var{b}),\var{e}):\var{C}(\var{e})$} \probe
\end{axiom}

\begin{axiom}{1.4.5} \axiomName{Coproduct computation left} \probe : 

\actualAxiom{$rule\_coproduct-comp1 : \ctx, \var{z}:\var{A}+\var{B} \vdash \var{C}:\U{i},\quad \ctx,\var{a}:\var{A} \vdash \var{c}:\var{C}(inl(\var{a})),\quad \ctx,\var{b}:\var{B} \vdash \var{d}:\var{C}(inr(\var{b})),\quad \ctx \vdash \var{x}:\var{A}\ \text{——}\ \ctx \vdash \text{let}\ \var{x} = \var{a}, \var{z} = inl(\var{a})\ \text{in}\ ind(\var{C}, \var{c}, \var{d}, inl(\var{a}))\equiv \var{c}(\var{x}):\var{C}(\var{z})$} \probe
\end{axiom}

\begin{axiom}{1.4.6} \axiomName{Coproduct computation right} \probe : 

\actualAxiom{$rule\_coproduct-comp1 : \ctx, \var{z}:\var{A}+\var{B} \vdash \var{C}:\U{i},\quad \ctx,\var{a}:\var{A} \vdash \var{c}:\var{C}(inl(\var{a})),\quad \ctx,\var{b}:\var{B} \vdash \var{d}:\var{C}(inr(\var{b})),\quad \ctx \vdash \var{y}:\var{B}\ \text{——}\ \ctx \vdash \text{let}\ \var{y} = \var{b}, \var{z} = inr(\var{b})\ \text{in}\ ind(\var{C}, \var{c}, \var{d}, inr(\var{b}))\equiv \var{d}(\var{y}):\var{C}(\var{z})$} \probe
\end{axiom}
